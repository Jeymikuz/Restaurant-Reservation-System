CREATE PROC Add_Reservation
@TableID int,
@FirstName nvarchar(30),
@LastName nvarchar(60),
@Email nvarchar(120),
@PhoneNumber nvarchar(20),
@ReservationDate datetime,
@ReservationLength int,
@Accepted bit
AS
INSERT Reservation(TableID, FirstName, LastName, Email, PhoneNumber,ReservationDate,ReservationLength,Accepted)
VALUES (@TableID, @FirstName, @LastName, @Email, @PhoneNumber,@ReservationDate,@ReservationLength,@Accepted)

UPDATE Reservation
SET ReservationExpire= DATEADD(HOUR,@ReservationLength,@ReservationDate)
GO


CREATE PROCEDURE Reservation_GetAll
AS
BEGIN
	SELECT *
	FROM Reservation

END
GO

CREATE PROCEDURE Reservation_GetByFirstName
@FirstName nvarchar(60)
AS
BEGIN
	SELECT *
	FROM Reservation
	WHERE FirstName = @FirstName;

END
GO

CREATE PROCEDURE Reservation_GetByLastName
@LastName nvarchar(60)
AS
BEGIN
	SELECT *
	FROM Reservation
	WHERE LastName = @LastName;

END
GO

CREATE PROCEDURE Reservation_GetByFirstAndLastName
@FirstName nvarchar(50),
@LastName nvarchar(60)
AS
BEGIN
	SELECT *
	FROM Reservation
	WHERE LastName = @LastName and FirstName= @FirstName;

END
GO



CREATE PROC Add_Table
@TableID int,
@ReservationID int,
@NumberOfChairs int
AS
INSERT ReservationTable(TableID,ReservationID,NumberOfChairs)
VALUES (@TableID,@ReservationID,@NumberOfChairs)
GO

CREATE PROCEDURE GetAllTables

	FROM ReservationTable
